    00x03 构建

    • 构建命令
    # 更新子模块
git submodule update --init --recursive

mkdir build
cd build
cmake -DWITH_BOOST=../boost_1_77_0  -DCMAKE_VERBOSE_MAKEFILE=ON -DBUILD_CONFIG=xtrabackup_release -DWITH_MAN_PAGES=OFF ..

make -j`nproc`
